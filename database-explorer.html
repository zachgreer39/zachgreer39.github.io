üîç Example Queries

Basic Data Exploration

Sample At-Bats
-- View sample at-bats with outcomes SELECT game_pk, inning, inning_topbot, final_event, batter, pitcher FROM tbl_atbats WHERE final_event IS NOT NULL LIMIT 15;
At-Bat Outcomes
-- Most common at-bat results SELECT final_event, COUNT(*) as occurrences, ROUND(COUNT(*) * 100.0 / (SELECT COUNT(*) FROM tbl_atbats WHERE final_event IS NOT NULL), 2) as percentage FROM tbl_atbats WHERE final_event IS NOT NULL GROUP BY final_event ORDER BY occurrences DESC;

Pitch Analysis

Pitch Types Analysis
-- Most common pitch types SELECT pitch_type, COUNT(*) as pitch_count, ROUND(AVG(release_speed), 1) as avg_speed FROM tbl_pitchinfo WHERE pitch_type IS NOT NULL GROUP BY pitch_type ORDER BY pitch_count DESC;
Count Situations
-- Pitch outcomes by count SELECT balls, strikes, description, COUNT(*) as frequency FROM tbl_pitchinfo WHERE description IS NOT NULL GROUP BY balls, strikes, description ORDER BY balls, strikes, frequency DESC;

Advanced Analysis

Join Analysis
-- First pitch type for each at-bat outcome SELECT ab.final_event, p.pitch_type, COUNT(*) as count FROM tbl_atbats ab JOIN tbl_pitchinfo p ON ab.game_pk = p.game_pk AND ab.at_bat_number = p.at_bat_number WHERE ab.final_event IS NOT NULL AND p.pitch_number = 1 AND p.pitch_type IS NOT NULL GROUP BY ab.final_event, p.pitch_type ORDER BY ab.final_event, count DESC;
Team Statistics
-- Home runs by team SELECT home_team, COUNT(*) as home_runs FROM tbl_atbats WHERE final_event = 'home_run' GROUP BY home_team ORDER BY home_runs DESC LIMIT 10;

üîÑ Recursive Queries

Count Generation
-- Generate all possible ball-strike counts recursively WITH RECURSIVE pitch_counts AS ( -- Base case: start with 0-0 count SELECT 0 as balls, 0 as strikes UNION ALL -- Recursive case: increment balls or strikes SELECT CASE WHEN balls < 3 THEN balls + 1 ELSE balls END as balls, CASE WHEN strikes < 2 THEN strikes + 1 ELSE strikes END as strikes FROM pitch_counts WHERE balls < 3 OR strikes < 2 ) SELECT DISTINCT balls, strikes, balls || '-' || strikes as count_display FROM pitch_counts ORDER BY balls, strikes;
Pitcher Game Progression
-- Track pitcher performance across games WITH RECURSIVE pitcher_games AS ( -- Base case: each pitcher's first game SELECT pitcher, MIN(game_pk) as game_pk, 1 as game_number FROM tbl_atbats GROUP BY pitcher UNION ALL -- Recursive case: next games for each pitcher SELECT ab.pitcher, ab.game_pk, pg.game_number + 1 FROM tbl_atbats ab JOIN pitcher_games pg ON ab.pitcher = pg.pitcher WHERE ab.game_pk > pg.game_pk AND pg.game_number < 5 GROUP BY ab.pitcher, ab.game_pk, pg.game_number ) SELECT pitcher, game_number, COUNT(*) as total_games FROM pitcher_games GROUP BY pitcher, game_number ORDER BY pitcher, game_number;
